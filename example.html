	showNum: 10, //当前显示条数

	}
});

属性 备注说明
current number，当前页
showNum number，当前显示条数
callback function(){}，回调函数

弹出框

弹框 
调用方式：
var modal = Modal({
	id:'modal',
	title:'提示',
	content:_html,
	complete:function(){
		// do something ...
	},
	confirm:function(){
		// do something ...
	},
	cancel:function(){
		// do something ...
	}
})

参数说明

属性 备注说明
id string，例如："sui-modal"
type number，0：弹出框，1：提示框
addClass string，添加类名，例如：addClass="a"，addClass="b"
title string，标题
width nubmer，例如：width=500
backdrop boolean值，true或者false
confirmButton boolean值，true或者false
cancelButton boolean值，true或者false
confirmText string，确定按钮名称
cancelText string，取消按钮名称
complete function(){}，打开弹框后回调
confirm function(){}，确定事件回调
cancel function(){}，取消事件回调

参数说明

属性 备注说明

hide string，隐藏工具提示，调用：$('[data-toggle="tip"]').tip('hide')
